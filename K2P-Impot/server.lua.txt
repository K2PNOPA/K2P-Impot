local ESX = exports["es_extended"]:getSharedObject()

local function PreleverImpot()
    local joueurs = ESX.GetPlayers()

    for _, playerId in pairs(joueurs) do
        local xPlayer = ESX.GetPlayerFromId(playerId)

        if xPlayer then
            local argent = xPlayer.getMoney()

            if argent >= Config.MontantImpot then
                xPlayer.removeMoney(Config.MontantImpot)

                TriggerClientEvent(
                    'esx:showNotification',
                    playerId,
                    ('💰 Impôt prélevé : %s$'):format(Config.MontantImpot)
                )

                print(("[IMPOT] %s a payé %s$"):format(
                    GetPlayerName(playerId),
                    Config.MontantImpot
                ))
            else
                TriggerClientEvent(
                    'esx:showNotification',
                    playerId,
                    ('❌ Vous n avez pas assez d argent pour payer %s$ d impôt.'):format(Config.MontantImpot)
                )

                print(("[IMPOT] %s n'a pas assez d'argent"):format(
                    GetPlayerName(playerId)
                ))
            end
        end
    end
end

CreateThread(function()
    print("^2[IMPOT]^7 Systeme d'impot demarre.")

    while true do
        Wait(Config.IntervalleImpot)
        PreleverImpot()
    end
end)
